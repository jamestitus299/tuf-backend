CREATE TABLE IF NOT EXISTS banneritems (
            id INT AUTO_INCREMENT PRIMARY KEY,
            title VARCHAR(255) NOT NULL,
            description TEXT,
            link VARCHAR(255),
            image_link VARCHAR(255),
            expiration_time DATETIME,
            created_time TIMESTAMP DEFAULT CURRENT_TIMESTAMP
        )
